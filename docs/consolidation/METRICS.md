### Script 1: File Count by Category

```bash
#!/bin/bash
# Count markdown files by category

echo "Documentation File Counts by Category"
echo "======================================"

for dir in architecture design operations testing phases reference reviews governance consolidation; do
    if [ -d "docs/$dir" ]; then
        count=$(find "docs/$dir" -name "*.md" -type f | wc -l | tr -d ' ')
```

### Script 4: Documentation Freshness

```bash
#!/bin/bash
# Check documentation freshness (last modified dates)

echo "Documentation Freshness Report"
echo "============================="

# Find files modified in last 7 days
week_old=$(find docs -name "*.md" -type f ! -path "*/archive/*" -mtime -7 | wc -l | tr -d ' ')
echo "Updated in last week: $week_old files"

# Find files modified in last 30 days
month_old=$(find docs -name "*.md" -type f ! -path "*/archive/*" -mtime -30 | wc -l | tr -d ' ')
echo "Updated in last month: $month_old files"

# Find files older than 180 days (6 months)
stale=$(find docs -name "*.md" -type f ! -path "*/archive/*" -mtime +180 | wc -l | tr -d ' ')
echo "Stale (>6 months): $stale files"

# Calculate total active files
total=$(find docs -name "*.md" -type f ! -path "*/archive/*" | wc -l | tr -d ' ')
echo ""
echo "Total active files: $total"
echo "Freshness: $(echo "scale=1; ($month_old * 100) / $total" | bc)% updated in last month"
```
